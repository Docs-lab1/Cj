const express = require('express');
const router = express.Router();
const auth = require('../middleware/auth');

// Get nearby vehicles
router.get('/nearby', auth, async (req, res) => {
  const db = req.app.get('db');
  const { lat, lng, radius = 5, vehicle_type } = req.query;
  
  try {
    let query = `
      SELECT 
        v.id, v.registration_number, v.vehicle_type, v.model, v.color,
        v.status, v.current_latitude, v.current_longitude,
        v.price_per_km, v.base_fare,
        u.full_name as driver_name, u.phone as driver_phone, u.rating as driver_rating,
        (6371 * acos(cos(radians($1)) * cos(radians(v.current_latitude)) * 
        cos(radians(v.current_longitude) - radians($2)) + sin(radians($1)) * 
        sin(radians(v.current_latitude)))) AS distance
      FROM vehicles v
      JOIN drivers d ON v.driver_id = d.id
      JOIN users u ON d.user_id = u.id
      WHERE v.is_active = true AND v.status = 'available' AND d.is_online = true
    `;
    
    const params = [lat, lng];
    
    if (vehicle_type && vehicle_type !== 'all') {
      query += ` AND v.vehicle_type = $${params.length + 1}`;
      params.push(vehicle_type);
    }
    
    query += ` HAVING (6371 * acos(cos(radians($1)) * cos(radians(v.current_latitude)) * 
              cos(radians(v.current_longitude) - radians($2)) + sin(radians($1)) * 
              sin(radians(v.current_latitude)))) < $${params.length + 1}
              ORDER BY distance LIMIT 30`;
    
    params.push(radius);
    
    const result = await db.query(query, params);
    
    res.json({
      vehicles: result.rows.map(v => ({
        ...v,
        distance_km: parseFloat(v.distance).toFixed(2),
        eta_minutes: Math.ceil(v.distance * 3)
      }))
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Failed to fetch nearby vehicles' });
  }
});

module.exports = router;
